136 inline void progressbar::update() {
137  if (n_cycles == 0)
138  throw std::runtime_error("progressbar::update: number of cycles not set");
139 
140  if (!update_is_called) {
141  if (do_show_bar == true) {
142  output << opening_bracket_char;
143  for (int _ = 0; _ < 50; _++) output << todo_char;
144  output << closing_bracket_char << " 0%";
145  } else {
146  output << "0%";
147  }
148  }
149  update_is_called = true;
150 
151  int perc = 0;
152 
153  // compute percentage, if did not change, do nothing and return
154  perc = static_cast<int>(progress * 100. / (n_cycles - 1));
155  if (perc < last_perc) return;
156 
157  // update percentage each unit
158  if (perc == last_perc + 1) {
159  // erase the correct number of characters
160  if (perc <= 10) {
161  output << "\b\b" << perc << '%';
162  } else if (perc > 10 && perc < 100) {
163  output << "\b\b\b" << perc << '%';
164  } else if (perc == 100) {
165  output << "\b\b\b" << perc << '%';
166  }
167  }
168  if (do_show_bar == true) {
169  // update bar every ten units
170  if (perc % 2 == 0) {
171  // erase closing bracket
172  output << std::string(closing_bracket_char.size(), '\b');
173  // erase trailing percentage characters
174  if (perc < 10)
175  output << "\b\b\b";
176  else if (perc >= 10 && perc < 100)
177  output << "\b\b\b\b";
178  else if (perc == 100)
179  output << "\b\b\b\b\b";
180 
181  // erase 'todo_char'
182  for (int j = 0; j < 50 - (perc - 1) / 2; ++j) {
183  output << std::string(todo_char.size(), '\b');
184  }
185 
186  // add one additional 'done_char'
187  if (perc == 0)
188  output << todo_char;
189  else
190  output << done_char;
191 
192  // refill with 'todo_char'
193  for (int j = 0; j < 50 - (perc - 1) / 2 - 1; ++j) output << todo_char;
194 
195  // readd trailing percentage characters
196  output << closing_bracket_char << ' ' << perc << '%';
197  }
198  }
199  last_perc = perc;
200  ++progress;
201  output << std::flush;
202 
203  return;
204 }
