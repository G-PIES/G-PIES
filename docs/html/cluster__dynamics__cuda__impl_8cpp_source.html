660 */
661 __CUDADECL__ gp_float ClusterDynamicsImpl::dislocation_promotion_probability(size_t n) const
662 {
663  gp_float dr = cluster_radius(n + 1) - cluster_radius(n);
664 
665  return (2. * cluster_radius(n) * dr + std::pow(dr, 2.))
666  / (std::pow(M_PI * mean_dislocation_radius_val / 2., 2) - std::pow(cluster_radius(n), 2.));
667 }

674 gp_float ClusterDynamicsImpl::dislocation_density_derivative() const
675 {
676  auto self = this->self;
677  gp_float gain = thrust::transform_reduce(
678  indices.begin(), indices.end(),
679  [self] __CUDADECL__ (const int& idx)
680  {
681  return self->cluster_radius(idx) * self->dislocation_promotion_probability(idx) * self->ii_absorption(idx) * self->interstitials[idx];
682  },
683  0.0, thrust::plus<gp_float>()
684  );
685 
686  gain *= 2. * M_PI / material.atomic_volume;
687 
688  return
689  gain
690  - reactor.dislocation_density_evolution *
691  std::pow(material.burgers_vector, 2.) *
692  std::pow(dislocation_density, 3. / 2.);
693 }
