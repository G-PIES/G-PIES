639 gp_float ClusterDynamicsCudaImpl::dislocation_density_derivative() const {
640  auto self = this->self;
641  gp_float gain = thrust::transform_reduce(
642  indices.begin(), indices.end(),
643  [self] __CUDADECL__(const int &idx) {
644  return self->cluster_radius(idx) *
645  self->i_dislocation_loop_unfault_probability(idx) *
646  self->ii_absorption(idx) * self->interstitials[idx];
647  },
648  0.0, thrust::plus<gp_float>());
649 
650  gain *= 2. * M_PI / material.atomic_volume;
651 
652  return gain - reactor.dislocation_density_evolution *
653  std::pow(material.burgers_vector, 2.) *
654  std::pow(dislocation_density, 3. / 2.);

739 int ClusterDynamicsCudaImpl::system([[maybe_unused]] double t,
740  N_Vector state_vector,
741  N_Vector derivatives_vector,
742  void *user_data) {

744  static_cast<ClusterDynamicsCudaImpl *>(user_data);
745 
746  gp_float *i_state = N_VGetArrayPointer(state_vector);
747  gp_float *v_state = i_state + cd->max_cluster_size + 2;
748  cd->dislocation_density = *(v_state + cd->max_cluster_size + 2);
749 
750  cd->step_init();
751 
752  thrust::copy(i_state, i_state + cd->max_cluster_size,
753  cd->interstitials.begin());
754  thrust::copy(v_state, v_state + cd->max_cluster_size, cd->vacancies.begin());
755 
756  cd->host_interstitials = cd->interstitials;
757  cd->host_vacancies = cd->vacancies;
758 
759  gp_float *i_derivatives = N_VGetArrayPointer(derivatives_vector);
760  gp_float *v_derivatives = i_derivatives + cd->max_cluster_size + 2;
761  gp_float *dislocation_derivative = v_derivatives + cd->max_cluster_size + 2;
762 
763  N_VConst(0.0, derivatives_vector);
764 
765  auto self = cd->self;
766  thrust::transform(cd->indices.begin() + 1, cd->indices.end(),
767  cd->device_i_derivatives.begin() + 2,
768  [self] __CUDADECL__(const int &idx) {
769  return self->i_concentration_derivative(idx);
770  });
771 
772  thrust::transform(cd->indices.begin() + 1, cd->indices.end(),
773  cd->device_v_derivatives.begin() + 2,
774  [self] __CUDADECL__(const int &idx) {
775  return self->v_concentration_derivative(idx);
776  });
777 
778  thrust::copy(cd->device_i_derivatives.begin(), cd->device_i_derivatives.end(),
779  i_derivatives);
780  thrust::copy(cd->device_v_derivatives.begin(), cd->device_v_derivatives.end(),
781  v_derivatives);
782 
783  i_derivatives[1] = cd->i1_concentration_derivative();
784  v_derivatives[1] = cd->v1_concentration_derivative();
785  *dislocation_derivative = cd->dislocation_density_derivative();
786 
787  return 0;
788 }

799 ClusterDynamicsCudaImpl::ClusterDynamicsCudaImpl(ClusterDynamicsConfig &config)
800  : time(0.0),
801  interstitials(config.max_cluster_size + 2, 0.0),
802  device_i_derivatives(config.max_cluster_size + 2, 0.0),
803  vacancies(config.max_cluster_size + 2, 0.0),
804  device_v_derivatives(config.max_cluster_size + 2, 0.0),
805  material(*config.material.impl()),
806  reactor(*config.reactor.impl()),
807  indices(config.max_cluster_size, 0),
808  host_interstitials(config.max_cluster_size + 1, 0.0),
809  host_vacancies(config.max_cluster_size + 1, 0.0) {
810  max_cluster_size = config.max_cluster_size;
811  data_validation_on = config.data_validation_on;
812  relative_tolerance = config.relative_tolerance;
813  absolute_tolerance = config.absolute_tolerance;
814  max_num_integration_steps = config.max_num_integration_steps;
815  min_integration_step = config.min_integration_step;
816  max_integration_step = config.max_integration_step;
817 
818  dislocation_density = material.dislocation_density_0;
819 
820  state_size = 2 * (max_cluster_size + 2) + 1;
821 
822  /* Create the SUNDIALS context */
823  int sunerr = SUNContext_Create(SUN_COMM_NULL, &sun_context);
824  if (sunerr)
825  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
826  ClusterDynamicsState());
827 
828  /* Create the initial state */
830  state = N_VNew_Serial(state_size, sun_context);
831 
832  /* Initialize State Values */
833  gp_float *i_state = N_VGetArrayPointer(state);
834  gp_float *v_state = i_state + max_cluster_size + 2;
835  for (size_t i = 0; i < max_cluster_size; ++i) {
836  i_state[i] = config.init_interstitials[i];
837  v_state[i] = config.init_vacancies[i];
838  }
839 
840  N_VGetArrayPointer(state)[state_size - 2] = material.dislocation_density_0;
841 
842  /* Call CVodeCreate to create the solver memory and specify the
843  * Backward Differentiation Formula */
844  cvodes_memory_block = CVodeCreate(CV_BDF, sun_context);
845 
846  /* Call CVodeInit to initialize the integrator memory and specify the
847  * user's right hand side function in y'=f(t,y), the initial time T0, and
848  * the initial dependent variable vector y. */
849  sunerr = CVodeInit(cvodes_memory_block, system, 0, state);
850  if (sunerr)
851  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
852  ClusterDynamicsState());
853 
854  /* Call CVodeSVtolerances to specify the scalar relative tolerance
855  * and scalar absolute tolerances */
856  sunerr = CVodeSStolerances(cvodes_memory_block, relative_tolerance,
857  absolute_tolerance);
858  if (sunerr)
859  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
860  ClusterDynamicsState());
861 
862  /* Create dense jacobian matrix */
863  jacobian_matrix = SUNDenseMatrix(state_size, state_size, sun_context);
864 
865  /* Create linear solver object for use by CVode */
866  linear_solver = SUNLinSol_Dense(state, jacobian_matrix, sun_context);
867 
868  sunerr = CVodeSetUserData(cvodes_memory_block, static_cast<void *>(this));
869  if (sunerr)
870  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
871  ClusterDynamicsState());
872 
873  sunerr = CVodeSetMaxNumSteps(cvodes_memory_block, max_num_integration_steps);
874  if (sunerr)
875  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
876  ClusterDynamicsState());
877 
878  sunerr = CVodeSetMinStep(cvodes_memory_block, min_integration_step);
879  if (sunerr)
880  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
881  ClusterDynamicsState());
882 
883  sunerr = CVodeSetMaxStep(cvodes_memory_block, max_integration_step);
884  if (sunerr)
885  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
886  ClusterDynamicsState());
887 
888  // TODO - set to time delta?
889  sunerr = CVodeSetInitStep(cvodes_memory_block, 1e-10);
890  if (sunerr)
891  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
892  ClusterDynamicsState());
893 
894  /* Attach the matrix and linear solver */
895  sunerr =
896  CVodeSetLinearSolver(cvodes_memory_block, linear_solver, jacobian_matrix);
897  if (sunerr)
898  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
899  ClusterDynamicsState());
900 
901  sunerr = CVodeSetInterpolateStopTime(cvodes_memory_block, SUNTRUE);
902  if (sunerr)
903  throw ClusterDynamicsException(SUNGetErrMsg(sunerr),
904  ClusterDynamicsState());
905 
906  ClusterDynamicsCudaImpl *raw_self;
907  cudaMalloc(&raw_self, sizeof(ClusterDynamicsCudaImpl));
908  cudaMemcpy(raw_self, this, sizeof(ClusterDynamicsCudaImpl),
909  cudaMemcpyHostToDevice);
910  self = thrust::device_ptr<ClusterDynamicsCudaImpl>(raw_self);
911  thrust::sequence(indices.begin(), indices.end(), 1);
912 }
